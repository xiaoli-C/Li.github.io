<!doctype html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<meta name="viewport" content="width=device-width,initial-scale=1,minimum-scale=1,maximum-scale=1,user-scalable=no" />
		<link href="css/mui.css" rel="stylesheet" />
	</head>
	
	<body>
		
		<!--插入值-->
		<div id="mv">
			<!--插值-->
			<span>{{msg}}</span><br/>
			<!--插一次性值-->
			<span v-once>{{oncevalue}}</span>
			<button v-on:click="changeValue">修改一次性值</button>
			<!--插入HTML-->
			<p v-html="html"></p>
			<!--插入标签属性，属性不能通过{{}}绑定，通过v-bind绑定-->
				<!--绑定size属性--><!----><!----><!---->
				<input v-bind:size="size"/>
				<!--绑定id属性-->
				<text v-bind:id="id"></text>
		</div>
		<br/><br/>
		
		<!--条件渲染-->
		<div id="mv1">
			<!--
				v-if
				v-else-if
				v-else
			-->
				<!--通过key值判断-->
				<p v-if="type === 'Chinese'">你好，世界！</p>
				<p v-else-if="type === 'English'">hello,world!</p>
				<p v-else>选择语言</p>
				<button v-on:click="exchangeC">中文</button>
				<button v-on:click="exchangeE">英文</button><br/>
				<br/>
				<!--<template> 元素当做不可见的包裹元素,在<template>中使用-->
				<button v-on:click="chooseShow">选择显示吗</button>
				<template v-if="show">
					<li>1</li>
					<li>2</li>
					<li>3</li>
				</template>
				<br/>
			<!--
				v-show,
				v-if 是惰性的：如果在初始渲染时条件为假，则什么也不做,
				——直到条件第一次变为真时，才会开始渲染条件块。

				v-show 不管初始条件是什么，元素总是会被渲染，
				并且只是简单地基于 CSS 进行切换。
				v-show 不支持 <template> 元素，也不支持 v-else
			-->
				<button v-on:click="csshow">show?</button>
				<h3 v-show="ok">show</h3>		
		</div>
		<br/><br/>
        
		<!--列表渲染-->
		<div id="mv2">
			<!-- v-for 绑定数组数据循环读出 -->
				<li v-for="arrdata in arr">{{arrdata}}</li>
				<!--访问所有父作用域的 property。
				v-for 还支持一个可选的第二个参数，即当前项的索引。-->
				<br/>
				<li v-for="(item,index) in items">{{parent}}-{{index+1}}-{{item.msg}}</li>
			
			<!--在v-for中使用对象-->
				<br/>
				<li v-for="value in object">{{value}}</li>
				<!--对象可以使用三个参数-->
				<li v-for="(value,name,index) in object">{{index}}-{{name}}:{{value}}</li>
		
			<!--在<template>上使用，可以渲染多个元素-->
			<template v-for="item in items">
			    <li>{{ item.msg }}</li>
			    <li>请填写</li>
			  </template>
		</div>
		<br/><br/>
		
		<!--事件监听-->
		<div id="mv3">
			<!--绑定点击事件-->
			<button v-on:click="sayhello">点击事件</button>
			<br/><br/>
			<!--方法传参-->
			<button v-on:click="say('hi')">sayhi</button>
			<button v-on:click="say('bye')">saybye</button>
			<!--
				v-on用来监听事件并绑定函数 
				@click 缩写
			-->  
			<button @click="say('come on')">saycomeon</button>
			<!--事件有修饰符，通过修饰符可以限定事件的条件
				事件修饰符、按键修饰符等。。。
			-->
		</div>
		<br/><br/>
		
		<!--双向绑定-->
		<div id="mv4">
			<!--用 v-model 指令在表单 <input>、<textarea> 及 <select> 元素上创建双向数据绑定。
			v-model 会忽略所有表单元素的 value、checked、selected attribute 的初始值而总是将 Vue 实例的数据作为数据来源
			-->
			<input v-model="msg" placeholder="edit me"/>
			<p>{{msg}}</p>
			<!--多行文本-->
			<span>Multiline message is:</span>
			<p style="white-space: pre-line;">{{message}}</p>
			<br>
			<textarea v-model="message" placeholder="add multiple lines"></textarea>
			<!--复选框-->
			<br/>
			<!--单复选框-->
			<input type="checkbox" id="checkbox" v-model="checked">
			<!--for 属性规定 label 与哪个表单元素绑定。-->
			<label for="checkbox">{{checked}}</label>
			<!--多复选框
				多个复选框通过v-model互相联系起来
				单选是通过v-model绑定到一个值
				多选是通过v-model绑定到一个数组-->
			<br/>
			<input type="checkbox" id="jack" value="Jack" v-model="checkedNames">
			<label for="jack">Jack</label>
			<input type="checkbox" id="john" value="John" v-model="checkedNames">
			<label for="john">John</label>
			<input type="checkbox" id="mike" value="Mike" v-model="checkedNames">
			<label for="mike">Mike</label>
			<br>
			<span>Checked names: {{ checkedNames }}</span>
			<br/><br/>
			<!--单选框
				多个单选组件通过v-model互相联系-->
			<input type="radio"  id="one" value="one" v-model="picked">
			<label for="one">one</label>
			<br/>
			<input type="radio" id="two" value="two" v-model="picked">
			<label for="two">two</label>
			<br/>
			<input type="radio" id="three" value="three" v-model="picked">
			<label for="three">three</label>
			<br/>
			<span>单选：{{picked}}</span>
			<br/><br/>
			<!--选择框
				单选时绑定到一个值
				多选时绑定到一个数组-->
			<select v-model="option" id="option">
				<option disabled value="">请选择</option>
				<option>A</option>
				<option>B</option>
				<option>C</option>
			</select>
			<span>选择：{{option}}</span>
			<br/>
			<!--多选选择框-->
			<select  id="options" multiple style="width: 50px">
				<option disabled value="">请选择</option>
				<option>A</option>
				<option>B</option>
				<option>C</option>
			</select>
			<span>选择：{{options}}</span>
			<br/>
		</div>
		
		
		
	
		 
		
		
		
		<!-- 引入Vue依赖 -->
		<script src="https://cdn.jsdelivr.net/npm/vue@2.6.14/dist/vue.js"></script>
		<!-- 引入axios文件 -->
		<script src="https://unpkg.com/axios/dist/axios.min.js"></script>
		<script>
			/* 声明Vue */ 
			var mv = new Vue({
				el:"#mv",
				data:{
					msg : "插入文本",
					oncevalue : "一次性插值",
					html : "<li>插入第一条HTML</li><li>插入第二条HTML</li>",
					size : 50
				},
				methods : {
					changeValue : function(){
						this.oncevalue ="改变一次性值"
					}
				}
			})
			
			var mv1 = new Vue({
				el : "#mv1",
				data : {
					type : "",
					show : true,
					ok : true
				},
				methods:{
					exchangeC : function(){
						this.type="Chinese"
					},
					exchangeE : function(){
						this.type="English"
					},
					chooseShow : function(){
						this.show=!this.show
					},
					csshow : function(){
						this.ok = !this.ok
					}
				}
			})
			
			var mv2 = new Vue({
				el : "#mv2",
				data : {
					arr : ["列表循环","a","b","c"],
					items : [{msg:"姓名"},{msg:"性别"},{msg:"身高"},{msg:"体重"}],
					parent : "同学",
					object : {
						title : "目录",
						chapter : "第一章",
						value : "第一节"
					}
				}
			})
			
			var mv3 = new Vue({
				el : "#mv3",
				data : {},
				methods:{
					sayhello : function(event){
						<!--`event` 是原生 DOM 事件-->
						alert(event);
						alert("hello!")
					},
					say : function(message){
						alert(message);
					}
				}
			})
			
			var mv4 = new Vue({
				el : "#mv4",
				data : {
					msg : "双向绑定",
					message : "",
					checked : "true",
					checkedNames : [],
					picked : '',
					option : '',
					options : []
				},
				methods : {
					
				}
			})
			
			
								
			
			
			
			/* axios */
			/* axios.get */
			axios.get("www.baidu.com",{
				parms : {
					id : 10,
					name : 'li'
				}
			}).then(res=>{
				console(res.data)
			})
			/* axios.post */
			axios.post("www.baidu.com",{
				id : 10,
				name : 'li'
			}).then(res=>{
				
			})
			
			/* axios() */
			axios({
				url: 'www.baidu.com',
				method: 'get',
				params: {id: 1}
			}).then(res=>{
				
			})
			
		</script>  
	</body>

</html>
