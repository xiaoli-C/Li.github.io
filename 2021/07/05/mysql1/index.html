一、Mysql概述

常用mysql命令行命令

<1.启动MYSQL服务 net start mysql

停止MYSQL服务 net stop mysql

<2 . 在命令行中登陆MYSQL控制台

语法格式 mysql –user=root –password=123456 或 mysql –uroot –p123456

<3 . 进入MYSQL命令行工具后 , 使用status; 或/s 查看运行环境信息

<4 .显示所有数据库 : show databases;

   <5 .创建属于我们自己的数据库:     create database DB_name;

<6 .切换数据库的语法 : use DB_name;

<7 .显示数据库中的所有表 : show tables;

<8.初始化数据 ：source sql脚本的绝对路径；

文件以sql结尾，称为“sql脚本”。当一个文件的扩展名是.sql，并且该文件中编写了大量的sql语句，我们称这样的文件为sql脚本。

​ 注意：直接使用source命令可以执行sql脚本。sql脚本中的数据量太大的时候，无法打开，请使用source命令完成初始化。

<9 .显示某个表创建时的全部信息 : show create table table_name;

<10 .查看表的具体属性信息及表中各字段的描述：Describe table_name; 缩写形式 : desc table_name;

<11.删除数据库 drop database bjpowernode;

三、MySQL分区

  1. 分区：数据库中的一个表，按照规则分为若干个小表，不影响业务逻辑。

  2. 分区的优点

    • 可以储存更多的数据
    • 优化查询，where包含分区条件时，可以只扫描几个分区表来提高效率，sum()等聚合函数在分区上处理更高效
    • 删除无效的数据通过删除相关的分区，效率更高
    • 分散数据，增加吞吐量

  3. 分区概述

    • 分区根据分区键来分区，分区键可以是区间值，特定值，Hash函数等
    • 同一个表的所有分区必须使用同一个储存引擎，不同表之间可以不同
    • 分区类型
      • RANGE
      • LIST
      • HASH
      • KEY
    • 5.1版本分区键类型必须是INT(KEY分区时可以使用BLOB、TEXT之外的类型)，5.5后RANGE和LIST支持非整数分区
    • 如果表有主键或唯一键，其必须包含分区键，即分区表不能用主键\唯一键之外的键作为分区键分区

  4. 各个分区

    • RANGE

      常用于分组删除和分组查询

      create table emp(
    empno varchar(20) not null ,
    empname varchar(20),
    deptno int,
    birthdate date,
    salary int
)
<!--分区的最后一个表的范围为maxvalue，因为如果指定3000，当增加数据大于3000时，则报错-->
partition by range(salary)(
    partition p1 values less than (1000),
    partition p2 values less than (2000),
    partition p3 values less than maxvalue
);

<!--在日期或字符串上进行分区，要使用函数转化-->
<!--日期转换函数
	year()
	to_days()
	to_seconds()-->
create table emp(
    empno varchar(20) not null ,
    empname varchar(20),
    deptno int,
    birthdate date not null,
	salary int
)
partition by range(year(birthdate))(
	partition p1 values less than (1980),
	partition p2 values less than (1990),
	partition p3 values less than maxvalue
);

    • LIST

      create table emp(
    empno  varchar(20) not null ,
    empname varchar(20),
    deptno  int,
    birthdate date not null,
    salary int
)
<!--通过枚举值来分区-->
partition by list(deptno)(
    partition p1 values in  (10),
    partition p2 values in  (20),
    partition p3 values  in  (30，40)
);

    • HASH

      • 两种分区：常规取模分区，线性二次幂分区
      create table emp(
    empno varchar(20) not null ,
	empname varchar(20),
	deptno int,
	birthdate date not null,
	salary int
)
partition by hash(year(birthdate))
partitions 4;
<!--线性，+linear-->
create table emp(
    empno varchar(20) not null ,
	empname varchar(20),
	deptno int,
	birthdate date not null,
	salary int
)
partition by linear hash(year(birthdate))
partitions 4;

    • KEY

      • key按照MySQL提供的hash函数分区
      • 若不指定，则主键为分区键，若无主键，非空唯一键为分区键
      create table emp(
    empno varchar(20) not null ,
	empname varchar(20),
 	deptno int,
	birthdate date not null,
	salary int
)
partition by key(birthdate)
partitions 4;

    • COLUMNS分区

      <!--
	整数类型：tinyint、smallint、mediumint、int、bigint
	日期时间类型：date、datetime
	字符类型：char、varchar、binary、varbinary,不支持text和blob
-->

<!--多列分区
	a,b两个分区键
-->
CREATE TABLE t_range_test (
    a INT,
    b INT 
)
PARTITION BY RANGE COLUMNS (a,b) (
    <!--
    	less than (num1,num2)
    	分区条件判断：a值小于num1 or a值等于num1,b值小于num2
    	(1,10)分在p2  1<10 or (1=10 and 10<10)
    	(10,9)分在p2  10<10 or (10=10 and 9<10)
    	(10,10)分在p3 10<10 or (10=10 and 10<20)
    -->
    PARTITION p1 VALUES LESS THAN  (0,10),
    PARTITION p2 VALUES LESS THAN  (10,10),
    PARTITION p3 VALUES LESS THAN  (10,20),
    PARTITION p4 VALUES LESS THAN  (10,MAXVALUE),
    PARTITION p4 VALUES LESS THAN  (MAXALUE,MAXVALUE)
);

    • 子分区

      CREATE TABLE ts (id INT, purchased DATE)
    PARTITION BY RANGE(YEAR(purchased))
    SUBPARTITION BY HASH(TO_DAYS(purchased))
    SUBPARTITIONS 2
    (
        PARTITION p0 VALUES LESS THAN (1990),
        PARTITION p1 VALUES LESS THAN (2000),
        PARTITION p2 VALUES LESS THAN MAXVALUE
    )；
 <!--
 	表ts 有3个RANGE分区。这3个分区中的每一个分区p0, p1, 和 p2,又被进一步分成了2个子分区。
 	P0分为两个子区，两个子区值均小于1990
 	p1分为两个子区，值均小于2000
 	p3分为两个子区，小于maxvalue
 -->

  5. null值得处理

    • RANGE: 当做最小值处理
    • LIST：Null值必须在枚举中写出，否则不接受
    • HASH\KEY: 当做零值处理

  6. 分区管理

    <!--删除分区
	删除 RANGE和LIST分区,删除分区之后，连同分区中的数据一同删除
	RANGE分区删除之后，增加数据后会分配到其他分区
	LIST分区删除之后，不能再添加已删除的分区键的数据
-->
alter table table_name drop partition p_num
eg：alter table ts drop partition p1

<!--增加分区
	RANGE：只能在最大端加
	alter table table_name add partition(partition p5 values less than (maxvalues)) 
	LIST
	alter table table_name add partition（partition p5 values in (2020)）
-->

<!--重新定义分区:保存原数据的情况下，同时改变结构
	重新定义只能定义相邻的分区，不能跨分区，同时重新定义的范围必须原分区覆盖相同的范围，也不能改变分区表的类型
-->
	alter table table_name reorganize partition into xxx
	
	RANGE:
	将p3(2005-2015)分为p2(2005-2010)p3(2010-2015)
	alter table ts reorganize partition p3 into (
        partition p2 values less than (2010)
        partition p3 values less than (2015)
    )
    将p2(2005-2010)p3(2010-2015)合并为一个P3(2005-1015)
    alter table ts reorganize partition p2,p3 into (
        partition p3 values less than (2015)
    )
    
    LIST:
    现有 p4 values in (5)
    	p5 values in (6)
    将p4 变为 p4 values in (5,7)
    	alter table ts add partition p6(partition values in (7))
    	alter table ts reorganize partition p4,p5,p6 into (
            partition p4 values in (5,7)
            partition p5 values in (6 )
        )
    <!--HASH和KEY分区管理-->
<!--不能通过删除来减少分区，只能通过合并来减少-->
alter table table_name coalesce partition 2；<!--分区将合成为2个-->
<!--增加分区-->
alter table table_name add partition 2;<!--原来只有两个分区，增加2个，现在是四个-->

三、存储过程和函数

  1. 概念

    存储过程和函数是预先编译并存储在数据库中的一段SQL的集合，区别在于函数必须有返回值，存储过程的参数可以是IN,OUT,INOUT而函数只能是IN。

  2. 相关命令操作

    • 储存过程的创建
      CREATE PROCEDURE 创建的存储过程名字(OUT|IN|INOUT 参数名 数据类型,…,…) 特征 过程体;

      • 特征值：
        • LANGUAGE SQL：表明过程体使用SQL语言
        • [NOT] DETERMINISTIC : 每次输入一样，输出也一样的程序
        • {CONTAINS SQL|NO SQL|READS SQL DATA|MODIFIES DATA}
          • CONTAINS SQL:子程序不包括读出后写的语句，默认值
          • NO SQL:自程序不包括SQL语句。
          • READS SQL DATA:包含读语句，不包含写
          • MPDIFIES DATA:包含写语句，不包括读语句
        • SQL SECURITY {DEFINER|INVOKER}
          • DEFINER 创建者许可执行
          • INVOKER 调用者许可执行
        • COMMENT：注释信息
        DELIMITER $$
create procedure GetScores(
	out minScore decimal(8,2),
	out avgScore decimal(8,2),
	out maxScore decimal(8,2)
)
begin
	select min(score) into minScore from user;
	select avg(score) into avgScore from user;
	select max(score) into maxScore from user;
end;
DELIMITER ;

    • 函数的创建

      CREATE FUNCTION 创建的存储函数名字(参数名称 参数类型,…,…) RETURNS 返回值得类型 函数体;

      DELIMITER $$
CREATE FUNCTION func_employee_sal (empno INT(11))   //传入值
    RETURNS DOUBLE(10,2)     //返回值
COMMENT'查询某个雇员的工资'
BEGIN
    RETURN (SELECT sal
        FROM t_employee
        WHERE t_employee.empno=empno);
END$$
DELIMITER ;

    • 删除

      drop procedure|function sp_name

    • 查看

      show procedure|function status 
show create procedure|function sp_name

    • 变量的使用 

      DECLARE var_name[,…] type [DEFAULT value]
SET var_name=expr[,…]

	declare prodDiscRate decimal(8,2);
	set prodDiscRate = 0.88;

    • 条件定义和处理

       <!---条件创建-->
 DECLARE condition_name CONDITION FOR [condition_type]
SQLSTATE [VALUE] sqlstate_value | mysql_erroe_code
 	condition_type：条件类型，用于定义MySQL的错误，SQLSTATE是长度为5的字符串类型的错误代码；
 	mysql_error_code是数值类型的错误代码。
 	例如ERROR 1142（42000），sql_state_value的值是'42000'，mysql_error_code的值是1142。
 	
 <!--条件处理-->
 DECLARE handler_type HANDLER FOR  condition_value[,...] sp_statement
（1）handler_type：指定错误处理方式
    CONTINUE：遇到错误不处理，继续执行
    EXIT：遇到错误立即退出
    UNDO：遇到错误撤回之前的操作
（2）condition_value：表示错误类型
    SQLSTATE [VALUE] sqlstate_value：包含5个字符的字符串错误值
    mysql_error_code：数值类型的错误代码
    cond_name：定义条件的名称
    SQLWARNING：匹配所有以01开头的SQLSTATE错误代码
    NOT FOUND：匹配所有以02开头的SQLSTATE错误代码
    SQL EXCEPTION：匹配所有没有被SQLWARNING或NOT FOUND捕获的SQLSTATE错误代码
（3）sp_statement
    程序语句段，表示在遇到定义的错误时执行的存储过程或函数。

四、触发器

  2. ss

    CREATE TRIGGER 创建的触发器名字 BEFFOR|AFTER  INSERT|UPDATE|DELETE

ON 表名字 FOR  触发器的执行间隔   触发器的SQL语句
    create trigger t1 
	after
	insert
	on ord
	for each row
		begin
				update goods set num=num-2 where gid = 1;
		end$
		
(1)查看已有触发器：show triggers 
   (2)删除已有触发器：drop trigger triggerName

  3. 触发器中引用行变量

    （1）在触发目标上执行insert操作后会有一个新行，如果在触发事件中需要用到这个新行的变量，可以用new关键字表示
    （2）在触发目标上执行delete操作后会有一个旧行，如果在触发事件中需要用到这个旧行的变量，可以用old关键字表示
    （3）在触发目标上执行update操作后原纪录是旧行，新记录是新行，可以使用new和old关键字来分别操作
